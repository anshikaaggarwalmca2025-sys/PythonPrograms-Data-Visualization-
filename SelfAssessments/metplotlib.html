import matplotlib.pyplot as plt
import numpy as np

# Data
cities = ['City A', 'City B', 'City C', 'City D']
population = [5, 8, 3, 10]  # in millions
area = [600, 800, 300, 1200]  # in km^2
gdp = [200, 500, 150, 700]  # in billion USD

# Plot
plt.figure(figsize=(8,5))
scatter = plt.scatter(area, population, s=np.array(gdp), alpha=0.5, c=np.array(gdp), cmap='viridis')
plt.colorbar(scatter, label="GDP (Billion USD)")

# Add city labels
for i, city in enumerate(cities):
    plt.text(area[i]+10, population[i]+0.1, city)

plt.xlabel("Area (km²)")
plt.ylabel("Population (millions)")
plt.title("Population vs Area (Bubble size = GDP)")
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# Data
classes = ['Class A', 'Class B', 'Class C']
scores = [
    np.random.randint(50, 100, 30),  # Class A
    np.random.randint(40, 95, 30),   # Class B
    np.random.randint(60, 100, 30)   # Class C
]

plt.figure(figsize=(8,5))

# Draw a simple violin-like plot using histograms
for i, score in enumerate(scores):
    hist, bins = np.histogram(score, bins=15, density=True)
    hist = hist / hist.max() * 0.4  # normalize and scale width
    bin_centers = (bins[:-1] + bins[1:]) / 2
    plt.fill_betweenx(bin_centers, i - hist, i + hist, alpha=0.5)
    plt.plot(np.full_like(bin_centers, i), bin_centers, color='black', linewidth=0.5)

plt.xticks(range(len(classes)), classes)
plt.title("Distribution of Exam Scores by Class (Approx Violin)")
plt.ylabel("Scores")
plt.xlim(-0.5, len(classes)-0.5)
plt.show()

import matplotlib.pyplot as plt

# Create figure and 2x2 subplots
fig, axs = plt.subplots(2, 2, figsize=(12,10))

# ---------------- Line Chart ----------------
x_line = [1, 2, 3, 4, 5]
y_line = [10, 20, 15, 25, 30]
axs[0,0].plot(x_line, y_line, marker='o', color='blue')
axs[0,0].set_title("Line Chart")
axs[0,0].set_xlabel("X-axis")
axs[0,0].set_ylabel("Y-axis")

# ---------------- Bar Chart ----------------
categories = ['A', 'B', 'C', 'D']
values = [5, 7, 3, 8]
axs[0,1].bar(categories, values, color='orange')
axs[0,1].set_title("Bar Chart")
axs[0,1].set_xlabel("Category")
axs[0,1].set_ylabel("Value")

# ---------------- Pie Chart ----------------
pie_sizes = [20, 30, 25, 25]
pie_labels = ['Q1', 'Q2', 'Q3', 'Q4']
axs[1,0].pie(pie_sizes, labels=pie_labels, autopct='%1.1f%%', explode=[0.1,0,0,0])
axs[1,0].set_title("Pie Chart")

# ---------------- Scatter Plot ----------------
x_scatter = [1, 2, 3, 4, 5]
y_scatter = [5, 10, 15, 20, 25]
axs[1,1].scatter(x_scatter, y_scatter, color='green')
axs[1,1].set_title("Scatter Plot")
axs[1,1].set_xlabel("X-axis")
axs[1,1].set_ylabel("Y-axis")

# Adjust layout
plt.tight_layout()
plt.show()
